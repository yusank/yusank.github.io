func (l *List) lIndexNode(i int) *listNode {
    if l.head == nil {
        return nil
    }

    // æ”¯æŒåå‘æŸ¥ï¼Œå³å¦‚æœ i å°äº 0 åˆ™è®¤ä¸ºæ˜¯å€’æ•°ç¬¬ i ä¸ªå…ƒç´ ï¼ŒæŠŠ i è¯¥ä¸ºæ­£æ•°ç¬¬ i ä¸ª

        return nil

        idx     int
        cur     = l.head
        reverse = i > l.length/2+1

    // ä¸ºäº†æŸ¥è¯¢æ•ˆç‡ï¼Œåšä¸€ä¸ªå°å°çš„ä¼˜åŒ–

        } else {
            idx++
            cur = cur.next
        }
    }

    return cur
}

func (l *List) LInsert(target, newValue string, flag int) bool {
    if l.head == nil {
        return false
    }

    // æ‰¾åˆ°å…ƒç´ 
    node := l.findNode(target)
    if node == nil {
        return false
    }

        node.value = newValue
        return true

    l.length++
    // insert after
    if flag > 0 {
        next := node.next
        node.next = newNode
        newNode.prev = node
        if next == nil {
            l.tail = newNode
        } else {
            newNode.next = next
            next.prev = newNode

    prev := node.prev
    node.prev = newNode
    newNode.next = node
    if prev == nil {
        l.head = newNode
    } else {
        newNode.prev = prev
        prev.next = newNode
    }

    return true
}

    )

    for head != nil && idx <= stop {
        if idx >= start {
            values = append(values, head.value)
        }

        idx++
        head = head.next
    }

    return
}
